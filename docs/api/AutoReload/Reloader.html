# File lib/autoreload/reloader.rb, line 40
def start
  update # primate the path loads

  @thread = Thread.new do
    loop do
      begin
        update
      rescue Exception
        warn 'update failed: ' + $!
      end
      sleep @interval
    end
  end
  @thread.abort_on_exception = true
end
